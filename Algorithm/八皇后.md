//判断该节点是否可以放入信息
bool isOk(int row,int column) {
    int i;

     int leftup = column - 1;
     int rightup = column + 1;

    for(i = row-1;i >=0; i--) {
        if (result[i] == column) {
            return false;
        }
        //判断该列是否有值

        //判断左上对角线是否有值
        if (leftup >=0) {
            if (result[i] == leftup) {
                return false;
            }
        }

        //判断右上对角线是否有值
        if (rightup < 8) {
            if (result[i] == rightup) {
                return false;
            }
        }

        leftup--;
        rightup++;
    }
    return true;
}

    int column;
