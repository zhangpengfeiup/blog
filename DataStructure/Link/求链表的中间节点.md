- 使用快慢指针寻找链表的中间节点
```
#include <stdio.h>
#include <stdlib.h>

struct ListNode {
    int data;
    struct ListNode *next;
};

struct ListNode* createListTail(int n) {
    struct ListNode *head,*p,*tmp;
    head = NULL;

    int i;
    for (i = 0;i < n;i++) {
         tmp = (struct ListNode *)malloc(sizeof(struct ListNode));
         scanf("%d",&tmp->data);
         // this is need init the node->next,otherwise the link can not stop.
         tmp->next = NULL;
        if (head == NULL) {
            head = tmp;
            p = head;
        }else {
            p->next = tmp;
            p = p->next;
        }
    }
    return head;
}

void printAllNode(struct ListNode *l) {
    if (l == NULL) {
        return NULL;
    }
    while (l) {
        printf("%d ",l->data);
        l = l->next;
    }
    printf("\n");
}

 // slow and quick point
struct ListNode* findMiddleOfLink(struct ListNode *L) {
    if (L == NULL) {
        return NULL;
    }
    struct ListNode *slow,*quick;
    slow = quick = L;
    while(slow->next && quick->next && quick->next->next) {
        slow = slow->next;
        quick = quick->next->next;
    }
    return slow;
}

int main()
{

    struct ListNode *l,*middle;
    l = createListTail(6);
    printAllNode(l);
    middle = findMiddleOfLink(l);
    if (middle == NULL) {
        printf("can not find\n");
    }else {
        printf("find it,it value is %d\n",middle->data);
    }
    return 0;
}
```
